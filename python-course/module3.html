Module 03

Control Flow & Functions

Loops, comprehensions, functions, closures, recursion, and memoization — the control structures that form every algorithm.

Loops

for Loop — Every Useful Form

# ── Basic iteration ───
for x in [1, 2, 3]:
    print(x)

# ── Range ───
for i in range(5):          # 0, 1, 2, 3, 4
for i in range(2, 8):       # 2, 3, 4, 5, 6, 7
for i in range(10, 0, -2):  # 10, 8, 6, 4, 2   (countdown, step -2)
for i in range(n-1, -1, -1):# n-1 down to 0   (reverse index loop)

# ── enumerate — index + value together ───
fruits = ["apple", "banana", "cherry"]
for i, fruit in enumerate(fruits):
    print(i, fruit)           # 0 apple, 1 banana, 2 cherry

for i, fruit in enumerate(fruits, start=1):  # start index at 1
    print(i, fruit)           # 1 apple, 2 banana, 3 cherry

# ── zip — iterate multiple lists in parallel ───
names  = ["Alice", "Bob", "Charlie"]
scores = [95, 87, 92]
for name, score in zip(names, scores):
    print(f"{name}: {score}")

# zip stops at the shortest list — use zip_longest for full coverage:
from itertools import zip_longest
for a, b in zip_longest([1,2,3], [4,5], fillvalue=0):
    print(a, b)   # (1,4) (2,5) (3,0)

# ── Nested loops ───
for i in range(rows):
    for j in range(cols):
        process(grid[i][j])

# ── for...else — runs if loop finishes WITHOUT break ───
target = 7
for x in nums:
    if x == target:
        print("found!")
        break
else:
    print("not found")   # only runs if no break occurred

while Loop

# Standard while
i = 0
while i < n:
    process(arr[i])
    i += 1

# Two-pointer idiom
left, right = 0, len(arr) - 1
while left < right:
    # ... do something with arr[left], arr[right]
    left += 1
    right -= 1

# Read until condition
while (line := input()):   # walrus operator := assigns AND evaluates
    process(line)

# while...else (same rule as for...else)
while stack:
    x = stack.pop()
    if x == target:
        break
else:
    print("never found target")

Comprehensions

Comprehensions are the most Python-idiomatic way to create collections. They are typically faster than equivalent for-loops because they are optimized at the bytecode level.

List Comprehension

# Syntax: [expression  for  variable  in  iterable  if  condition]

# Basic
squares = [x**2 for x in range(10)]         # [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]

# With filter
evens = [x for x in range(20) if x % 2 == 0]  # [0, 2, 4, ..., 18]

# Transformation + filter
result = [x**2 for x in nums if x > 0]

# Flatten 2D list
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
flat = [x for row in matrix for x in row]   # [1, 2, 3, 4, 5, 6, 7, 8, 9]
# Note: outer loop first, inner loop second (reads like: "for each row, for each x in row")

# Transpose matrix
transposed = [[row[i] for row in matrix] for i in range(len(matrix[0]))]

# Conditional expression inside comprehension
labels = ["pos" if x > 0 else "neg" for x in nums]

Dict and Set Comprehension

# Dict comprehension
freq = {c: "hello".count(c) for c in "hello"}  # {'h':1, 'e':1, 'l':2, 'o':1}

# Invert a dict (assumes values are unique)
d = {"a": 1, "b": 2, "c": 3}
inv = {v: k for k, v in d.items()}   # {1:'a', 2:'b', 3:'c'}

# Set comprehension
unique_lengths = {len(word) for word in ["cat", "dog", "elephant", "fox"]}
# {3, 8} — deduplicated

# Generator expression (lazy — does not build the full list)
# Use inside functions that consume iterables one at a time
total = sum(x**2 for x in range(1000))      # no intermediate list
any_neg = any(x < 0 for x in nums)          # short-circuits on first True
all_pos = all(x > 0 for x in nums)          # short-circuits on first False
max_sq  = max(x**2 for x in nums)

Functions

Parameters and Return Values

# Basic function
def add(a, b):
    return a + b

# Default arguments (evaluated ONCE at definition time)
def greet(name, prefix="Hello"):
    return f"{prefix}, {name}!"

greet("Alice")            # "Hello, Alice!"
greet("Bob", "Hi")        # "Hi, Bob!"
greet("Eve", prefix="Hey")# "Hey, Eve!" — keyword argument

# !! Mutable default argument TRAP
def append_to(x, result=[]):   # BAD — [] is created once and reused
    result.append(x)
    return result

append_to(1)   # [1]
append_to(2)   # [1, 2]  ← NOT [2]! Same list is reused!

# Fix: use None as sentinel
def append_to(x, result=None):
    if result is None:
        result = []
    result.append(x)
    return result

# Multiple return values (returns a tuple)
def min_max(arr):
    return min(arr), max(arr)

lo, hi = min_max([3, 1, 4, 1, 5])  # lo=1, hi=5

# *args — variable positional arguments (collected as tuple)
def total(*nums):
    return sum(nums)
total(1, 2, 3, 4)   # 10

# **kwargs — variable keyword arguments (collected as dict)
def show(**info):
    for k, v in info.items():
        print(f"{k} = {v}")
show(name="Alice", age=30)

# Unpacking into function calls
nums = [1, 2, 3]
print(*nums)         # print(1, 2, 3) — unpacks list as positional args
d = {"end": "\n"}
print("hi", **d)     # print("hi", end="\n") — unpacks dict as keyword args

Lambda Functions

# lambda: anonymous function, single expression
square = lambda x: x**2
square(5)   # 25

# Primary use: sort keys
nums = [3, -1, 4, -1, 5]
nums.sort(key=lambda x: abs(x))        # sort by absolute value
nums.sort(key=lambda x: (x%2, x))      # odds first, then evens; each group sorted

pairs = [(1,3), (2,1), (3,2)]
pairs.sort(key=lambda p: p[1])         # sort by second element
pairs.sort(key=lambda p: (-p[1], p[0]))# desc by second, then asc by first

# Sorting with multiple tie-breaking criteria (very common in intervals)
intervals = [(1,4), (1,2), (2,3), (1,3)]
intervals.sort(key=lambda x: (x[0], x[1]))  # start asc, then end asc

# Using lambda with filter/map (usually prefer comprehensions)
evens = list(filter(lambda x: x%2==0, nums))   # prefer: [x for x in nums if x%2==0]
doubled = list(map(lambda x: x*2, nums))        # prefer: [x*2 for x in nums]

Closures and Nested Functions

A closure is a function that captures variables from its enclosing scope. In coding problems, nested functions are useful for DFS/backtracking where you want to share state without global variables.

# Nested function captures outer variables (closure)
def find_paths(grid):
    rows, cols = len(grid), len(grid[0])
    result = []           # captured by inner function

    def dfs(r, c, path):
        if r == rows-1 and c == cols-1:
            result.append(path[:])   # result is from outer scope
            return
        for dr, dc in [(0,1),(1,0)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < rows and 0 <= nc < cols:
                path.append(grid[nr][nc])
                dfs(nr, nc, path)
                path.pop()

    dfs(0, 0, [grid[0][0]])
    return result

# nonlocal — modify a variable in enclosing (non-global) scope
def count_nodes(root):
    count = [0]             # trick: use list to avoid nonlocal
    def dfs(node):
        if not node: return
        count[0] += 1
        dfs(node.left)
        dfs(node.right)
    dfs(root)
    return count[0]

# Cleaner with nonlocal (Python 3)
def count_nodes(root):
    count = 0
    def dfs(node):
        nonlocal count       # tells Python we want the outer 'count'
        if not node: return
        count += 1
        dfs(node.left)
        dfs(node.right)
    dfs(root)
    return count

Recursion

A recursive function calls itself with a simpler subproblem until it reaches a base case. Every recursive call consumes stack space — Python's default limit is ~1000 frames.

Call stack for factorial(4)
factorial(4) └─ return 4 * factorial(3) └─ return 3 * factorial(2) └─ return 2 * factorial(1) └─ return 1 * factorial(0) └─ BASE CASE: return 1 Unwinds: 1 → 1*1=1 → 2*1=2 → 3*2=6 → 4*6=24 
# Three requirements for correct recursion:
# 1. Base case (stops the recursion)
# 2. Recursive case (moves toward base case)
# 3. Trust the recursion (assume recursive call returns correct result)

def factorial(n):
    if n <= 1:       # 1. base case
        return 1
    return n * factorial(n - 1)  # 2. recursive case, 3. trust it returns (n-1)!

# Fibonacci — naive (exponential time, don't use without memoization)
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)   # ← recomputes fib(n-2) multiple times!

# Increase recursion limit for deep problems
import sys
sys.setrecursionlimit(200000)   # needed for n up to ~100,000

# Convert recursion to iteration to avoid stack overflow
def factorial_iterative(n):
    result = 1
    for i in range(2, n+1):
        result *= i
    return result

Memoization — Caching Recursive Results

Memoization stores the result of a function call so that if the same arguments are seen again, we return the cached result immediately instead of recomputing. This transforms overlapping-subproblem recursion from exponential to polynomial time.

Without vs. with memoization for fib(5)
WITHOUT memo: WITH memo: fib(5) fib(5) ├─ fib(4) ├─ fib(4) │ ├─ fib(3) │ ├─ fib(3) │ │ ├─ fib(2) │ │ ├─ fib(2) │ │ │ ├─ fib(1) ✓ │ │ │ ├─ fib(1) ✓ │ │ │ └─ fib(0) ✓ │ │ │ └─ fib(0) ✓ │ │ └─ fib(1) ✓ │ │ └─ fib(1) ✓ (cached) │ └─ fib(3) ← recomputes! │ └─ fib(2) ✓ (cached) │ ├─ fib(2) ← recomputes! └─ fib(3) ✓ (cached) │ ... Total calls: 9 Total calls: 15 (grows as 2^n) Time: O(n), Space: O(n) 
from functools import lru_cache, cache

# @cache (Python 3.9+) — unbounded memoization, simplest to use
@cache
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)

fib(100)   # fast — each value computed once and cached

# @lru_cache(maxsize=None) — same as @cache, available from Python 3.2
@lru_cache(maxsize=None)
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)

# @lru_cache(maxsize=128) — bounded cache (LRU eviction)
# Use when memory is a concern

# Multiple parameters
@cache
def dp(i, j, state):
    if i == 0: return 0
    # ... transition
    return min(dp(i-1, j, state), dp(i, j-1, state))

# Cache with mutable arguments — use tuples, not lists
# lru_cache requires HASHABLE arguments
@cache
def dp(i, mask):           # mask is an int — hashable ✓
    ...

# If you need to pass a list, convert to tuple first
def solve(nums):
    @cache
    def dp(i, remaining):   # remaining is a tuple — hashable ✓
        ...
    return dp(0, tuple(nums))

# Clear cache between test cases (if reusing the function)
fib.cache_clear()
print(fib.cache_info())   # CacheInfo(hits=..., misses=..., maxsize=None, currsize=...)
💡
Top-down vs Bottom-up DP

Memoization (top-down) lets you write the natural recursive solution and just add @cache. Bottom-up DP (filling a table) avoids recursion overhead and stack overflow. For most problems, start with @cache to verify correctness, then convert to bottom-up if needed for performance.
